%T Test local vimrc

\usemodule[vim]

\startvimrc[name=bash]
let g:is_bash=1
\stopvimrc

\definevimtyping[SH][syntax=sh]
\definevimtyping[BASH][syntax=sh, vimrc=bash]

\starttext

\subject{SH default}
\startSH
  export foo=bar
\stopSH

\subject{BASH default} 
\startBASH
  export foo=bar
\stopBASH

\subject{BASH vimrc=none}
\startBASH[vimrc=none]
  export foo=bar
\stopBASH

\subject{SH vimrc=bash}
\startSH[vimrc=bash]
  export foo=bar
\stopSH

\stoptext

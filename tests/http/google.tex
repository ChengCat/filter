\usemodule[filter]

\defineexternalfilter
  [GoogleCharts]
  [filter={http://chart.apis.google.com/chart?},
   separator={&},
   method=httpget,
   readcommand=\ExternalPng]

\def\ExternalPng#1{\externalfigure[#1][method=png,scale=3000]}

\enabletrackers[third.externalfilter]
\enabletrackers[resolvers.schemes]

\starttext
\startGoogleCharts
  cht=bhg
  chs=550x230
  chd=t:100,50,115,80
  chxt=x,y
  chxl=1:|Python|Java|Ruby|.NET
  chxr=0,0,120
  chds=0,120
  chco=4D89F9
  chbh=35,0,15
  chg=8.33,0,5,5
\stopGoogleCharts

\stoptext
